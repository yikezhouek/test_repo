```r
Trips %>% 
  mutate(new_date = as_date(sdate)) %>% 
  inner_join(top_trip_dates, 
             by = c("sstation", "new_date")) %>% 
  mutate(days = wday(new_date, label = TRUE)) %>% 
  group_by(days, client) %>% 
  summarize(num = n()) %>% 
  group_by(client) %>% 
  mutate(prop = num/sum(num)) %>% 
  pivot_wider(id_cols = days,
              names_from = client,
              values_from = prop)
```
